-- [ PRIMARY KEY AND AUTO INCREMENT- EX: IT CAN AUTOMATICALLY INCREASE THE ID VALUE LIKE 1,2,3 ETC... WE DON'T HAVE TO DEFINE EACH AND EVERYTIME ID INSTEAD OF THE AUTO INCREMENT WILL HELP US ]
use dk;

create table transactions(
	t_id int,
    amount decimal(6,2)
);

alter table transactions
add constraint primary key(t_id);

alter table transactions
modify t_id int auto_increment;


insert into transactions (amount)
values (  4000.23 );

insert into transactions (amount)
values (  4000.23 ), (5000.20), (3000.22), (6000.00);

select * from transactions;

desc transactions;

drop table transactions;
