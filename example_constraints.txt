-- [ EXAMPLE OF CONSTRAINTS ]

use dk;

create table product(
	p_id int,
    p_name varchar(15) ,
    price decimal(6,2)

);

desc product;

drop table product;

-- [UNIQUE KEY]
alter table product
add constraint unique key (p_name);

-- [NOT NULL]
alter table product
modify p_name varchar(15) not null; -- NOT NULL is a column property, not a table constraint. WE MUST USE MODIFY

-- [CHECK]
alter table product
add constraint check_price check(price < 1000);  -- FOR CHECK CONSTRAINT WE MUST USE REFFERENCE NAME SO THAT WE CAN DROP THE CONSTRAINT.

-- [DEFAULT]
alter table product
modify price decimal(6,2) default 20; -- DEFAULT is a column property, not a table constraint. WE MUST USE MODIFY

drop constraint  check_price;
